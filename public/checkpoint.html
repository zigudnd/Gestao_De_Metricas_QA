Animações & Transições

/* Entrada de tab */ @keyframes fadeIn { from { opacity: 0; transform: translateY(5px); } to { opacity: 1; transform: translateY(0); } } /* Entrada da Home */ @keyframes homeIn { from { opacity: 0; transform: translateY(10px); } to { opacity: 1; transform: translateY(0); } } /* Hover de card */ .kpi-card:hover { transform: translateY(-3px); box-shadow: var(--shadow-md); } .sprint-card:hover { transform: translateY(-3px); } .btn-primary:hover { transform: translateY(-1px); } /* Modal */ .modal-overlay { transition: opacity 0.25s, visibility 0.25s; } .modal-content { transition: transform 0.25s ease; } /* Padrão global */ transition: all 0.2s ease; /* usado em botões, cards, links */ transition: all 0.15s; /* elementos menores (filtros, delete btn) */
